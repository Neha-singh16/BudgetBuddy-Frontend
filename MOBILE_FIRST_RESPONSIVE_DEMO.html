1. Update App.jsx

2. Update index.html Viewport

/* Mobile-first spacing */ .card { padding: var(--spacing-lg); /* 1.5rem on mobile */ gap: var(--spacing-md); /* 1rem */ } /* Responsive grid */ .card-grid { grid-template-columns: 1fr; /* 1 col mobile */ } @media (min-width: 768px) { .card-grid { grid-template-columns: repeat(2, 1fr); } } @media (min-width: 1024px) { .card-grid { grid-template-columns: repeat(3, 1fr); } }
